---
title: 'Get analysis'
description: 'Retrieve the stored AI analysis for a film, including all content layers.'
api: 'GET /api/analisis/{peliculaId}'
---

Returns the existing analysis record for the specified film, including all `capas_analisis` rows. If no analysis has been generated yet, the endpoint returns `404`.

<Info>
  Authentication is optional. Guest users may read existing analyses without a token.
</Info>

## Path parameters

<ParamField path="peliculaId" type="number" required>
  The internal Magic Film ID of the film whose analysis you want to retrieve.
</ParamField>

## Response

<ResponseField name="analisis" type="object" required>
  <Expandable title="properties" defaultOpen={true}>
    <ResponseField name="id" type="number">
      Internal analysis ID.
    </ResponseField>
    <ResponseField name="pelicula_id" type="number">
      The film this analysis belongs to.
    </ResponseField>
    <ResponseField name="tipo" type="string">
      Analysis depth. One of `entretenimiento` or `profundo`.
    </ResponseField>
    <ResponseField name="modelo_ia" type="string">
      The AI model used. Defaults to `gemini-1.5-flash`.
    </ResponseField>
    <ResponseField name="estado" type="string">
      Generation status. One of `pendiente`, `generando`, `completo`, or `error`.
    </ResponseField>
    <ResponseField name="idioma" type="string">
      Language code of the generated content (e.g., `es`).
    </ResponseField>
    <ResponseField name="eliminado" type="boolean">
      Whether the analysis has been soft-deleted.
    </ResponseField>
    <ResponseField name="generado_en" type="string">
      ISO 8601 timestamp of initial generation.
    </ResponseField>
    <ResponseField name="actualizado_en" type="string">
      ISO 8601 timestamp of last update.
    </ResponseField>
    <ResponseField name="capas" type="object[]">
      Ordered array of analysis layers generated by the AI.
      <Expandable title="Capa object properties" defaultOpen={true}>
        <ResponseField name="id" type="number">
          Internal ID of the layer.
        </ResponseField>
        <ResponseField name="nombre_capa" type="string">
          Layer identifier. One of the following enum values:

          | Value | Description |
          |---|---|
          | `resumen` | Plot summary |
          | `momentos` | Key moments |
          | `easter_eggs` | Hidden references and Easter eggs |
          | `curiosidades` | Behind-the-scenes trivia |
          | `narrativa` | Narrative structure analysis |
          | `simbolismo` | Symbolism and motif breakdown |
          | `contexto` | Historical or cultural context |
          | `tecnica` | Cinematography and technical analysis |
          | `conclusion` | Overall conclusion |
        </ResponseField>
        <ResponseField name="modo" type="string">
          The analysis mode this layer belongs to. One of `entretenimiento` or `profundo`.
        </ResponseField>
        <ResponseField name="contenido" type="string">
          Full text content of the layer, generated by the AI model.
        </ResponseField>
        <ResponseField name="orden" type="number">
          Display order within the analysis (ascending).
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
